演算子 意味 (Meaning) 使用例
== 等価（型変換あり） "5" == 5 → true

演算子 意味 (Meaning) 使用例
&& AND（かつ） true && false → false
|| OR（または） true || false → true
! NOT（否定） !true → false

// 論理演算子の短絡評価
const result = value1 || value2;  // value1がtruthyならvalue1、そうでなければvalue2

// Null合体演算子（ES2020）
const name = user.name ?? "ゲスト";  // nullまたはundefinedの場合のみデフォルト値

    return `こんにちは、${name}さん！`;
};

// アロー関数（省略形）
const greet = name => `こんにちは、${name}さん！`;

const greet = (name = "ゲスト") => {
    return `こんにちは、${name}さん！`;
};

// 可変長引数
const sum = (...numbers) => {
    return numbers.reduce((a, b) => a + b, 0);
};

クラス

クラスの定義と継承

// クラスの定義
class Person {
    // コンストラクタ
    constructor(name, age) {
        this.name = name;
        this.age = age;
    }
    
    // メソッド
    greet() {
        console.log(`こんにちは、${this.name}です`);
    }
    
    // ゲッター
    get info() {
        return `${this.name} (${this.age}歳)`;
    }
    
    // セッター
    set info(value) {
        [this.name, this.age] = value.split(",");
    }
    
    // 静的メソッド
    static create(name, age) {
        return new Person(name, age);
    }
}

// 継承
class Student extends Person {
    constructor(name, age, grade) {
        super(name, age);  // 親クラスのコンストラクタ
        this.grade = grade;
    }
    
    study() {
        console.log(`${this.name}は勉強中です`);
    }
}

        return data;

        console.error("エラー:", error);
    }
}

// 並列処理

try {
    // エラーが発生する可能性のあるコード
    const result = riskyOperation();
    console.log(result);
} catch (error) {

正規表現

正規表現の基本

パターン 意味 (Meaning) 例
. 任意の1文字 /a.c/ → "abc", "a1c"
* 0回以上 /ab*/ → "a", "ab", "abbb"
+ 1回以上 /ab+/ → "ab", "abbb"
? 0回または1回 /ab?/ → "a", "ab"
^ 行頭 /^Hello/
$ 行末 /world$/
[abc] 文字クラス /[aeiou]/ → 母音
\d 数字 /\d{3}/ → 3桁の数字
\w 英数字とアンダースコア /\w+/
\s 空白文字 /\s+/ 
// 正規表現の使用例
const pattern = /^\d{3}-\d{4}$/;  // 郵便番号パターン
const isValid = pattern.test("123-4567");

// 文字列の置換
const text = "Hello World";
const newText = text.replace(/World/, "JavaScript");

// グループのキャプチャ
const match = "2024-01-15".match(/(\d{4})-(\d{2})-(\d{2})/);
const [full, year, month, day] = match;
